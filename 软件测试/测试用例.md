public class DecodeHelp {
    public static void main(String[] args){
        char c1 = 'R';
        char c2 = 'o';
        char c3 = 'v';
        char c4 = 'z';
        final int KEY = 10;
        c1 = (char)(c1-KEY);
        c2 = (char)(c2-KEY);
        c3 = (char)(c3-KEY);
        c4 = (char)(c4-KEY);
        System.out.println(c1);
        System.out.println(c2);
        System.out.println(c3);
        System.out.println(c4);
    }
}
